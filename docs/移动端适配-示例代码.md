## 🔧 工具函数

### JavaScript辅助函数

```javascript
// px转rem
function pxToRem(px) {
  const fontSize = parseFloat(
    getComputedStyle(document.documentElement).fontSize
  );
  return px / fontSize;
}

// rem转px
function remToPx(rem) {
  const fontSize = parseFloat(
    getComputedStyle(document.documentElement).fontSize
  );
  return rem * fontSize;
}

// 判断是否移动端
function isMobile() {
  return document.documentElement.getAttribute('data-device') === 'mobile';
}

// 获取安全区域
function getSafeArea() {
  return {
    top: getComputedStyle(document.documentElement)
      .getPropertyValue('--safe-area-inset-top') || '0px',
    bottom: getComputedStyle(document.documentElement)
      .getPropertyValue('--safe-area-inset-bottom') || '0px'
  };
}
```
